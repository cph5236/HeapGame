# Tutorial Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add an auto-launching, replayable, hands-on tutorial that walks new players through Heap's full in-run gameplay loop on a tiny easy heap, then drops them into a real run.

**Architecture:** A dedicated `TutorialScene` builds a small fixed-fixture heap with all abilities granted, one rat, and one item. A pure `TutorialDirector` step machine gates progression on detected player actions; a `TutorialOverlay` shows pause-on-read popups and a Skip button. First-run is detected via a new `SaveData.tutorialDone` flag in `BootScene`.

**Tech Stack:** Phaser 3.90, TypeScript 5.9, Vitest. Spec: `docs/superpowers/specs/2026-06-17-tutorial-design.md`.

```ts
export type PlayerAction =
  | 'move' | 'jump' | 'walljump' | 'dash' | 'dive'
  | 'stomp' | 'pickup' | 'placeBlock';

export interface TutorialStep {
  id: string;
  message: string;
  advanceOn: 'tap' | PlayerAction;
  mode: 'info' | 'hint';
}

export interface TutorialCallbacks {
  onStepEnter(step: TutorialStep): void;
  onComplete(): void;
}

/** Pure, Phaser-agnostic step machine. The scene reacts to onStepEnter (keyed by
 *  step.id for side effects, step.mode for overlay style) and onComplete. */
export class TutorialDirector {
  private index = -1;
  private done = false;

  constructor(
    private readonly steps: TutorialStep[],
    private readonly callbacks: TutorialCallbacks,
  ) {}

  get currentStep(): TutorialStep | null {
    return this.index >= 0 && this.index < this.steps.length ? this.steps[this.index] : null;
  }

  start(): void {
    if (this.done) return;
    this.index = 0;
    this.enterCurrentOrComplete();
  }

  tapNext(): void {
    const step = this.currentStep;
    if (this.done || !step || step.advanceOn !== 'tap') return;
    this.advance();
  }

  notify(action: PlayerAction): void {
    const step = this.currentStep;
    if (this.done || !step || step.advanceOn !== action) return;
    this.advance();
  }

  skip(): void {
    if (this.done) return;
    this.complete();
  }

  private advance(): void {
    this.index += 1;
    this.enterCurrentOrComplete();
  }

  private enterCurrentOrComplete(): void {
    const step = this.currentStep;
    if (step) this.callbacks.onStepEnter(step);
    else this.complete();
  }

  private complete(): void {
    if (this.done) return;
    this.done = true;
    this.callbacks.onComplete();
  }
}
```

```ts
    else                      { this.unfreeze(); this.overlay.showHint(step.message); }
```
